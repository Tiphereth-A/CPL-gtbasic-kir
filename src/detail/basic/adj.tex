\begin{frame}{相邻与关联}
	\only<1->{在无向图 \(G = (V, E)\) 中, 若点 \(v\) 是边 \(e\) 的一个端点, 则称 \(v\) 和 \(e\) 是\textbf{相关联的}}

	\only<2->{对于两顶点 \(u\) 和 \(v\), 若存在边 \((u, v)\), 则称 \(u\) 和 \(v\) 是\textbf{相邻的}}

	\only<3->{一个顶点 \(v \in V\) 的\textbf{邻域}是所有与之相邻的顶点所构成的集合, 记作 \(N(v)\)}

	\only<4->{一个点集 \(S\) 的邻域是所有与 \(S\) 中至少一个点相邻的点所构成的集合, 记作 \(N(S)\), 即:

		\[
			N(S) = \bigcup_{v \in S} N(v)
		\]}
\end{frame}
