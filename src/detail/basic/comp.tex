\begin{frame}{补图}
	对于无向简单图 \(G = (V, E)\), 它的\textbf{补图}记作 \(\overline{G}\), 满足 \(V \left(\overline{G} \right) = V \left( G \right)\), 且对任意结点对 \((u, v)\), \((u, v) \in E \left(\overline{G} \right)\) 当且仅当 \((u, v) \notin E \left( G \right)\)
\end{frame}


\begin{frame}{反图}
	\only<1->{对于有向图 \(G = (V, E)\), 它的\textbf{反图}指的是点集不变, 每条边反向得到的图}

	\only<2->{即: 若 \(G\) 的反图为 \(G'=(V, E')\), 则 \(E'=\{(v, u):(u, v)\in E\}\)}
\end{frame}
